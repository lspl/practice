## 判断链表是否是回文结构
&ensp;&ensp;&ensp;&ensp;目前知道三种方法
- 方法一：将链表中所有元素压入栈。再将栈中的元素逐个弹出，和链表的元素进行比较
- 方法二：将链表后半部分元素压入栈。再将栈中元素逐个弹出，和链表的元素进行比较
- 方法三：将链表后半部分元素逆序，然后从链表两端向中间推进比较。最后将后半部分元素再逆序回来（原则上不能改变原来链表的结构）

直接上代码：
```java
import java.util.Stack;

public class Main {
    public static void main(String[] args) {
//        int[] arr = {1, 4, 2, 5, 0};
//        int[] arr = {1, 4, 2, 5, 0, 5, 2, 4, 1};
        int[] arr = {1, 4, 2, 5, 0, 0, 5, 2, 4, 1};
//        int[] arr = {1, 4, 2, 5, 0, 5, 2, 2, 1};
//        int[] arr = {1, 4};
//        int[] arr = {1, 4, 4, 1};
//        int[] arr = {1, 4, 1};
        ListNode list = ListNode.toListNode(arr);
        System.out.println(isPalindrome3(list));
    }

    // 方法一：将链表中所有元素压入栈。再将栈中的元素逐个弹出，和链表的元素进行比较
    public static boolean isPalindrome1(ListNode list) {
        if (list == null || list.next == null) {
            return true;
        }
        Stack<Integer> stack = new Stack<Integer>();
        ListNode temp = list;
        while (temp != null) {
            stack.push(temp.data);
            temp = temp.next;
        }
        temp = list;
        while (!stack.isEmpty()) {
            if (stack.pop() != temp.data) {
                return false;
            }
            temp = temp.next;
        }
        return true;
    }

    // 方法二：将链表后半部分元素压入栈。再将栈中元素逐个弹出，和链表的元素进行比较
    public static boolean isPalindrome2(ListNode list) {
        if (list == null || list.next == null) {
            return true;
        }
        ListNode slow = list;
        ListNode fast = list.next;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }

        Stack<Integer> stack = new Stack<Integer>();
        while ((slow = slow.next) != null) {
            stack.push(slow.data);
        }
        fast = list;
        while (!stack.isEmpty()) {
            if (stack.pop() != fast.data) {
                return false;
            }
            fast = fast.next;
        }
        return true;
    }

    // 方法三：将链表后半部分元素逆序，然后从链表两端向中间推进比较。最后将后半部分元素再逆序回来
    public static boolean isPalindrome3(ListNode list) {
        if (list == null || list.next == null) {
            return true;
        }

        if (list.next.next == null) {
            if (list.data != list.next.data) {
                return false;
            }
            return true;
        }
        ListNode slow = list;
        ListNode fast = list.next;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
        }

        ListNode newHead = reverseListNode(slow.next);
        fast = list;
        while (fast != null && newHead != null && !(fast.next == newHead || fast.next == newHead.next)) {
            if (fast.data != newHead.data) {
                return false;
            }
            fast = fast.next;
            newHead = newHead.next;
        }

        slow.next = reverseListNode(newHead);
        return true;
    }

    // 链表的逆序
    public static ListNode reverseListNode(ListNode list) {
        if (list == null || list.next == null) {
            return list;
        }
        ListNode pre = null, cur = list, next = list.next;
        while (cur != null) {
            next = cur.next;
            cur.next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
}

class ListNode {
    int data;
    ListNode next;

    public ListNode(int data) {
        this.data = data;
        this.next = null;
    }

    // 将数组转化为链表
    public static ListNode toListNode(int[] arr) {
        ListNode head = new ListNode(arr[0]);
        ListNode temp = head;
        for (int i = 1; i < arr.length; i++) {
            temp.next = new ListNode(arr[i]);
            temp = temp.next;
        }
        return head;
    }

    // 打印链表
    public static void println(ListNode list) {
        ListNode head = list;
        if (head != null) {
            System.out.print(head.data);
        }
        while ((head = head.next) != null) {
            System.out.print(" " + head.data);
        }
        System.out.println();
    }
}
```
